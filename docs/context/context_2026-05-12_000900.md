+extern "C" int32_t rc_server_start(uint16_t port, const uint8_t* psk,
+                                    uint32_t psk_length) {
+  if (!valid_psk(psk, psk_length)) {
+    return -1;
+  }
+  const uint16_t listen_port = port == 0U ? kDefaultTcpPort : port;
+  const rc_socket_t listener =
+      rc::transport::TcpTransport::listen_on(listen_port, kListenBacklog);
+  if (listener == kInvalidSocket) {
+    return -2;
+  }
+
+  rc::transport::TcpTransport accepted =
+      rc::transport::TcpTransport::accept_from(listener);
+  rc::transport::close_socket(listener);
+  if (accepted.socket_handle() == kInvalidSocket) {
+    return -3;
+  }
+
+  rc::transport::TlsTransport tls(std::move(accepted), rc::transport::TlsMode::kServer,
+                                  psk, psk_length);
+  return tls.handshake() ? 0 : -4;
+}
+
+extern "C" int32_t rc_client_connect(const char* ip_address, uint16_t port,
+                                      const uint8_t* psk,
+                                      uint32_t psk_length) {
+  if (ip_address == nullptr || !valid_psk(psk, psk_length)) {
+    return -1;
+  }
+  const uint16_t connect_port = port == 0U ? kDefaultTcpPort : port;
+  rc::transport::TlsTransport tls(rc::transport::TcpTransport(),
+                                  rc::transport::TlsMode::kClient, psk,
+                                  psk_length);
+  return tls.connect(ip_address, connect_port) ? 0 : -2;
+}
+
+extern "C" int32_t rc_discovery_start(uint16_t port, const char* device_name) {
